    icon = GFX_holy_order_jesuits
    trigger = {
        NOT = { current_age = age_of_empire }

        }

    }
    ai_priority = {
        factor = 2
        modifier = {
            factor = 2.0
            NOT = { religion = ROOT }
        }
    }
}

        }
    }
    cost = 50
    cost_type = dip_power
    per_province_effect = {

        factor = 1
        modifier = {
            factor = 2.0
            trade_goods = slaves
        }
        modifier = {
            factor = 2.0
            NOT = { culture = ROOT }
        }
    }
}
