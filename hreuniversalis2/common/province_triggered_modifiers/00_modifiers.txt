		normal_or_historical_nations = yes

		owner = {
			NOT = { religion = catholic }
			NOT = { religion = catholique }

				religion_group = christian
				religion_group = muslim
			}
		}
	}

	prestige = 0.5
	missionaries = 1
	
	#Province scope
	on_activation = {
	
	}
	
	on_deactivation = {
	
	}
	
}

	potential = {
	}
	
	#viewer = {

		if = {

			limit = {
				province_id = 202 # Avignon
			}

		else_if = {
			limit = {
				province_id = 781 # Praha
			}
			religion = hussite
		}
		else_if = {
			limit = {

			limit = {

	}

	local_missionary_strength = -0.05
