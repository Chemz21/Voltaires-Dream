#Fuggers
country_event = {
	id = vn_flavor_hab.4
	title = "vn_flavor_hab.4.t"
	desc = "vn_flavor_hab.4.d"
	picture = ADVISOR_eventPicture

	fire_only_once = yes

	trigger = {
		dynasty = "von Habsburg"
		is_year = 1500
		NOT = { started_in = 1504.1.1 }
		burgau_area = {
			owned_by = ROOT
			is_core = ROOT
		}
		capital_scope = { NOT = { area = burgau_area } }
		num_of_cities = 20
		NOT = { exists = BGX }
		normal_or_historical_nations = yes
	}

	mean_time_to_happen =  {
		months = 200
		modifier = {
			factor = 0.5
			is_emperor = yes
		}
		modifier = {
			factor = 0.75
			has_estate_privilege = estate_burghers_fugger_family
		}
	}

	option = {
		name = "vn_flavor_hab.4.a"
		ai_chance = { factor = 90 }
		add_imperial_influence = 10
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		burgau_area = {
			limit = { owned_by = ROOT }
			add_core = BGX
			remove_core = ROOT
		}
		release = BGX
		BGX = {
			add_opinion = { who = ROOT modifier = supported_independence }
			change_government = monarchy
			add_government_reform = autocracy_reform
			define_ruler = { dynasty = "Fugger" }
			set_government_rank = 1
		}
	}

	option = {
		name = "vn_flavor_hab.4.b"
		ai_chance = { factor = 10 }
		add_prestige = -5
	}
}
