###APULIA###
#Change primary culture to Sicilian?
country_event = {
	id = vn_missions.3
	title = vn_missions.3.t
	desc = vn_missions.3.d
	picture = KINGS_CHOICE_eventPicture

	is_triggered_only = yes

	option = {
		name = vn_missions.3.a
		ai_chance = { factor = 75 }
		add_stability_or_adm_power = yes
		every_owned_province = {
			limit = { region = sicilia_region }
			change_culture = ROOT
		}
	}
	option = {
		name = vn_missions.3.b
		ai_chance = { factor = 25 }
		change_primary_culture = sicilian
		set_ruler_culture = sicilian
		if = { limit = { has_heir = yes } set_heir_culture = sicilian }
		if = { limit = { has_consort = yes } set_consort_culture = sicilian }
		every_owned_province = {
			limit = { region = sicilia_region }
			add_unrest = -10
		}
	}
}

	id = vn_missions.4
	title = vn_missions.4.t
	desc = vn_missions.4.d
	picture = ICONS_IN_CHURCH_eventPicture

	is_triggered_only = yes

	option = {
		name = "vn_missions.4.a"
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0.5
				num_of_rebel_armies = 1
			}
		}
		add_treasury = 250
		every_owned_province = {
			limit = { religion = orthodox }
			add_unrest = 5
		}
	}
	option = {
		name = "vn_missions.4.b"
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.5
				num_of_loans = 1
			}
			modifier = {
				factor = 0
				is_bankrupt = yes
			}
		}
		add_country_modifier = {
			name = lem_mollify_clergy_mission
			duration = 9125
		}
	}
}
