				}
			}
		}
		effect = {

						name = "ara_carthago_delenda_est_mission"
						duration = 18250
					}
				}
			}
			else_if = {
                limit = {
                    tunisia_region = {
                        type = all
                        country_or_non_sovereign_subject_holds = ROOT
                    }
                }

                    limit = { country_or_non_sovereign_subject_holds = ROOT }

						name = "ara_carthago_delenda_est_mission2"
