#Fix Tribal Ownership
country_event = {
	id = fbievents.152
	title = none
	desc = none
	picture = BATTLE_eventPicture

	mean_time_to_happen = { days = 1 }
	hidden = yes

	trigger = {
		has_dlc = "Conquest of Paradise"
		any_tribal_land = {
			tribal_land_of = ROOT
			is_city = no
		}
	}

	option = {
		name = OK
		every_tribal_land_province = {
			limit = {
				tribal_land_of = ROOT
				is_city = no
			}
			add_core = ROOT
			cede_province = ROOT
		}
	}
}
