carmelite_order = {
    icon = GFX_holy_order_carmelites
    trigger = {
        NOT = { current_age = age_of_empire }
        OR = {
            tag = KOJ
			tag = ATC
			tag = TPI
			tag = EDE
        }
		OR = {
            religion = catholic
            religion = catholique
        }
    }
    cost = 50
    cost_type = adm_power
    per_province_effect = {
        add_base_tax = 1
    }
    per_province_abandon_effect = {
        add_base_tax = -1
    }
    modifier = {
        local_institution_spread = 0.1
		local_development_cost = -0.05
    }
    ai_priority = {
        factor = 2
        modifier = {
            factor = 2.0
            NOT = { religion = ROOT }
        }
    }
}
