#hidden event firing from losing a battle
province_event = {
	id = vn_flavor_sco.1
	title = none
	desc = none
	picture = BATTLE_eventPicture

	hidden = yes
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		FROM = {
			OR = {
				tag = NOL
				tag = SCY
				tag = FFE
			}
			war_with = HSC
			HSC = { has_ruler = "MacBheatha" }
		}
		has_ruler_leader_from = HSC
		NOT = { started_in = 1055.1.1 }
		NOT = { is_year = 1066 }
		normal_or_historical_nations = yes
	}

	immediate = {
		hidden_effect = {
			HSC = { country_event = { id = vn_flavor_sco.2 } }
			NOL = { country_event = { id = vn_flavor_sco.3 } }
		}
	}

	option = {
		name = "OK"
	}
}

country_event = {
	id = vn_flavor_sco.2
	title = "vn_flavor_sco.2.t"
	desc = "vn_flavor_sco.2.d"
	picture = BATTLE_eventPicture

	is_triggered_only = yes
	major = yes
	major_trigger = {
		capital_scope = {
			OR = {
				superregion = british_isles_superregion
				superregion = irish_superregion
			}
		}
	}

	option = {
		name = "vn_flavor_sco.2.a"
		tooltip = { kill_ruler = yes }
		if = {
			limit = { has_heir = yes }
			tooltip = { remove_heir = {} }
		}
		add_stability = -1
		define_ruler = {
			name = "Calum III"
			dynasty = "Dùn Chailleann"
			age = 30
			adm = 5
			dip = 3
			mil = 4
		}
		define_heir = {
			hidden = yes
			dynasty = "Dùn Chailleann"
			block_disinherit = yes
		}
		if = {
			limit = {
				any_owned_province = {
					is_core = SCY
				}
			}
			every_owned_province = {
				limit = { is_core = SCY }
				cede_province = SCY
			}	
		}
		tooltip = {
			white_peace = NOL
		}
	}
}
