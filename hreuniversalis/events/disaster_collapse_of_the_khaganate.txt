country_event = {
	id = collapse_of_the_khaganate.10
	title = "collapse_of_the_khaganate.10.t"
	desc = "collapse_of_the_khaganate.10.d"
	picture = CIVIL_WAR_eventPicture
	goto = mongol_pretender_prov
	is_triggered_only = yes

	trigger = {
		any_owned_province = {
			any_core_country = {
				NOT = { tag = ROOT }
				OR = {
					tag = GLH
					tag = KAZ
					tag = AST
					tag = CRI
					tag = NOG
					tag = KZH
					tag = JYA
					tag = OIR
					tag = AKK
					tag = QAR
				}
				exists = no
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					any_core_country = {
						NOT = { tag = ROOT }
						OR = {
							tag = GLH
							tag = KAZ
							tag = AST
							tag = CRI
							tag = NOG
							tag = KZH
							tag = JYA
							tag = OIR
							tag = AKK
							tag = QAR
						}
						exists = no
					}
				}
				save_event_target_as = mongol_pretender_prov
				random_core_country = {
					limit = {
						NOT = { tag = ROOT }
						OR = {
							tag = GLH
							tag = KAZ
							tag = AST
							tag = CRI
							tag = NOG
							tag = KZH
							tag = JYA
							tag = OIR
							tag = AKK
							tag = QAR
						}
						exists = no
					}
					save_event_target_as = mongol_pretender_tag
					ROOT = { release = PREV }

					#government
					change_government = tribal
					if = {
						limit = {
							OR = {
								tag = QAR
								tag = AKK
								tag = JYA
								tag = OIR
							}
						}
						add_government_reform = tribal_federation
					}
					else = {
						add_government_reform = steppe_horde
					}

					#ruler
					if = {
						limit = { tag = QAR }
						define_ruler = { dynasty = "Qara Qoyunlu" }
					}
					else_if = {
						limit = { tag = AKK }
						define_ruler = { dynasty = "Aq Qoyunlu" }
					}
					else_if = {
						limit = { tag = JYA }
						define_ruler = { dynasty = "Jalair" }
					}
					else = {
						define_ruler = { dynasty = "Borjigin" }
					}

					#for everyone
					if = {
						limit = { is_part_of_hre = yes }
						set_in_empire = no
					}
					add_treasury = 300
					add_manpower = 10
					set_government_rank = 5
					event_target:mongol_pretender_prov = {
						build_to_forcelimit = {
							infantry = 0.6
							cavalry = 0.4
						}
					}
				}
			}
		}
	}
	
	option = {
		name = collapse_of_the_khaganate.10.a
		custom_tooltip = collapse_of_the_khaganate.tt
		hidden_effect = {
			event_target:mongol_pretender_tag = {
				every_owned_province = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
		}
		ai_chance = {
			factor = 100
		}
	}
	
	option = {
		name = collapse_of_the_khaganate.10.b
		declare_war_with_cb = {
			who = event_target:mongol_pretender_tag
			casus_belli = cb_tribal_feud
		}
		ai_chance = {
			factor = 0
		}
	}

	option = {
		name = collapse_of_the_khaganate.10.c
		ai_chance = { factor = 0 }
		trigger = {	ai = no }
		hidden_effect = {
			event_target:mongol_pretender_tag = {
				every_owned_province = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
		}
		switch_tag = event_target:mongol_pretender_tag
	}
