estate_mamluks_disaster = {
	potential = {
		has_estate = estate_mamluks
		estate_influence = {
			estate = estate_mamluks
			influence = 80
		}
		OR = {
			NOT = { has_country_flag = mamluks_estate_triggered }
			had_country_flag = {
				flag = mamluks_estate_triggered
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_mamluks
			influence = 100
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_mamluks
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 1
	}
	
	progress = {
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_mamluks
				influence = 100
			}
		}
		modifier = {
			factor = 5
			tag = MAM
			NOT = { has_reform = mamluk_government }
		}		
	}
	
	can_end = {
		always = yes
	}

	on_start = estate_disasters.21
	
	on_start_effect = {
		if = {
			limit = { has_dlc = "The Cossacks" }
			custom_tooltip = estate_mamluks_disaster_tooltip_cossacks
		}
		else = {
			custom_tooltip = estate_mamluks_disaster_tooltip
		}
	}
	
	on_monthly = {
	}
}
