		if = {

					id = vn_flavor_swa.2

			factor = 10
			modifier = {
				factor = 2

			}
		}
		if = {
			limit = { exists = XBG }
			XBG = {
				country_event = {
					id = vn_flavor_swa.2
				}
			}

			factor = 1
		}

			culture = swabian
			hide_skills = yes
		}
		if = {
			limit = { has_heir = yes }
			remove_heir = {}
		}
		if = {
			limit = {
				LFB = {
					valid_for_personal_unions_trigger = yes
				}
			}

				LFB = {
					country_event = {
						id = vn_flavor_swa.5
					}
				}
			}

				who = XBG

			}

				type = cb_fabricated_claims

#answer for Swabia (good)
country_event = {
	id = vn_flavor_swa.3
	title = "vn_flavor_swa.3.t"
	desc = "vn_flavor_swa.3.d"
	picture = GOOD_WITH_MONARCH_eventPicture

	is_triggered_only = yes

	major_trigger = {
		is_part_of_hre = yes
	}

	immediate = {
		hidden_effect = {
			FROM = {
				if = {
					limit = {
						has_heir = no
					}
					define_heir = {} # prevent PU
				}
				exile_ruler_as = {
					name = rudolf_von_rheinfelden
				}
			}
			set_ruler = rudolf_von_rheinfelden
			define_heir = {}
		}
	}

	option = {
		name = vn_flavor_swa.3.a
		add_prestige = 10
		reverse_add_opinion = {
			who = FROM
			modifier = opinion_good_relations
		}
	}
}

				modifier = opinion_good_relations
			}

			who = FROM
			modifier = opinion_displeased
		}
	}
}
