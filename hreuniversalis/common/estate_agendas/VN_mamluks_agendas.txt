					}

					has_building = cathedral
				}
			}
			else_if = {
				limit = {
					check_variable = {
						which = estate_mamluks_build_temple_in_y_var
						value = 2
					}
				}
				event_target:agenda_province = {
					OR = {
						has_building = cathedral
						has_building = temple
					}
				}
			}
			else = {
				event_target:agenda_province = {
					OR = {
						has_building = monument
						has_building = temple
						has_building = cathedral
					}
				}
			}
		}
    }
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_mamluks
		}
		add_estate_loyalty = {
			estate = estate_mamluks
			loyalty = 10
		}
		if = {
			limit = {
				NOT = { crown_land_share = 15 }
				estate_territory = {
					estate = estate_mamluks
					territory = 10
				}
			}
			take_estate_land_share_small = { estate = estate_mamluks }
		}
		else = {
			event_target:agenda_province = {
				add_base_tax = 1
			}
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_mamluks_displeased
					value = 5
				}
			}
		}
		add_estate_loyalty_modifier = {
			estate = estate_mamluks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
    }
}

		factor = 1
	}

			is_state_core = root
		}
	}
	pre_effect = {
		set_variable = {
			which = develop_x_category
			value = 0
		}
		random_list = {
			2 = { #Mil is weighted higher
				set_variable = {
					which = develop_x_category
					value = 3
				}
			}
			1 = { #dip
				set_variable = {
					which = develop_x_category
					value = 2
				}
			}
			1 = { #adm
				set_variable = {
