fragmentation_of_poland = {
	potential = {
		NOT = { has_global_flag = had_fragmentation_of_poland }
		tag = POL
		NOT = { current_age = age_of_dynasticism }
		normal_or_historical_nations = yes
		has_reform = feudal_monarchy
	}

	can_start = {
		has_any_disaster = no
		current_age = age_of_holy_wars
		calc_true_if = {
			SDZ = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			WKP = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			KSZ = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			KRA = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			SIR = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			MAZ = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			SIL = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			amount = 4
		}
		has_heir = yes
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = { 
				OR = { 
					government = monarchy 
					has_reform = feudal_monarchy
				}
			}
			NOT = { tag = POL }
			current_age = age_of_dynasticism
		}
	}
		
	progress = {
		modifier = {
			factor = 0.5
			dynasty = "Piast"
		}
		modifier = {
			factor = 1
			has_heir = yes
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 0.5
			average_unrest = 1
		}
		modifier = {
			factor = 1
			NOT = { legitimacy_equivalent = 75 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 50 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 25 }
		}
		modifier = {
			factor = 1
			NOT = {
				any_owned_province = {
					OR = {
						region = hinterpommern_region
						region = westpreussen_region
					}
				}
			}
		}
	}
	
	modifier = {
		legitimacy = -1
		diplomatic_reputation = -2
	}	
	
	can_end = {
		tag = POL
		custom_trigger_tooltip = {
			tooltip = had_disaster_for_ten_years
			had_country_flag = {
				flag = poland_fragmentation_active
				days = 3650
			}
		}
		stability = 1
		calc_true_if = {
			OR = {
				MAZ = {
					exists = no
					is_subject_of = ROOT
				}
				NOT = {
					poland_superregion = {
						owned_by = MAZ
					}
				}
			}
			OR = {
				SIL = {
					exists = no
					is_subject_of = ROOT
				}
				NOT = {
					poland_superregion = {
						owned_by = SIL
					}
				}
			}
			OR = {
				WKP = {
					exists = no
					is_subject_of = ROOT
				}
				NOT = {
					poland_superregion = {
						owned_by = WKP
					}
				}
			}
			OR = {
				SDZ = {
					exists = no
					is_subject_of = ROOT
				}
				NOT = {
					poland_superregion = {
						owned_by = SDZ
					}
				}
			}
			amount = 3
		}
	}
	
	on_start = fragmentation_of_poland.1 #Last Will and Testament of [Root.Monarch.GetName]
	on_end = fragmentation_of_poland.100	#End of the Fragmentation Period

		events = {
		}
