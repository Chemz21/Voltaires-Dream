	title = vn_missions.3.t
	desc = vn_missions.3.d
	picture = KINGS_CHOICE_eventPicture

	is_triggered_only = yes

		name = vn_missions.3.a

			limit = { region = sicilia_region }
			change_culture = ROOT
		}
	}
	option = {
		name = vn_missions.3.b
		ai_chance = { factor = 25 }
		change_primary_culture = sicilian
		set_ruler_culture = sicilian
		if = { limit = { has_heir = yes } set_heir_culture = sicilian }
		if = { limit = { has_consort = yes } set_consort_culture = sicilian }
		every_owned_province = {
			limit = { region = sicilia_region }
			add_unrest = -10
		}
	}
}
